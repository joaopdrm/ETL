  ``etl_project/
  ├── dags/
  │   ├── etl_dag.py
  ├── data/

  │   ├── processed/
  ├── scripts/
  │   ├── extract.py
  │   ├── transform.py
  │   ├── load.py
  ├── config/
  │   ├── airflow.cfg
  │   ├── mongo_config.py
  ├── requirements.txt
  └── README.md``
